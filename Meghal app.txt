<script>

setTimeout(()=>{
document.getElementById("loader").style.display="none"
document.getElementById("app").style.display="block"
},3000)

let points = 0
let level = 1

function addPoints(p){

points += p
level = Math.floor(points/10)+1

document.getElementById("level").innerText =
"Adventure Level: " + level + " | Points: " + points

if(level >= 10){
document.getElementById("secret").innerText =
"🎉 SECRET LEVEL UNLOCKED\nThe AI recommends Meghal celebrate with juice at a cute café.\nRecommended companion: the guy who built this machine."
}

}

const jokes = [
{q:"Why don't eggs tell jokes?",a:"They'd crack each other up."},
{q:"Why did the coffee file a police report?",a:"It got mugged."},
{q:"Why did the bicycle fall over?",a:"It was two tired."},
{q:"Why can't your nose be 12 inches long?",a:"Because then it would be a foot."},
{q:"Why did the tomato turn red?",a:"Because it saw the salad dressing."}
]

let current = null

function newJoke(){

current = jokes[Math.floor(Math.random()*jokes.length)]

document.getElementById("joke").innerText = current.q
document.getElementById("answer").innerText = ""
document.getElementById("revealBtn").style.display="inline-block"

addPoints(1)

}

function reveal(){
document.getElementById("answer").innerText = current.a
}

const wheel = [
"Explore a new street",
"Find a cute café",
"Drink fresh juice",
"Plan a weekend trip",
"Take photos around the city",
"Try a new egg dish"
]

function spinWheel(){

let result = wheel[Math.floor(Math.random()*wheel.length)]

let wheelDiv = document.getElementById("wheel")

wheelDiv.classList.add("spin")

setTimeout(()=>{
wheelDiv.classList.remove("spin")
wheelDiv.innerText = "Adventure: " + result
},800)

addPoints(2)

}

const cafes = [
"Plant café with lots of sunlight ☀️",
"Hidden street café ☕",
"Juice bar with tropical drinks 🍹",
"Café with amazing egg sandwiches 🍳"
]

function cafe(){

let place = cafes[Math.floor(Math.random()*cafes.length)]

document.getElementById("cafe").innerText =
"Suggestion for Meghal: " + place

addPoints(2)

}

const destinations = [
"Tokyo 🇯🇵",
"Paris 🇫🇷",
"Bali 🇮🇩",
"Barcelona 🇪🇸",
"Rome 🇮🇹"
]

}

function dateChance(){

let percent = Math.floor(Math.random()*51)+50

document.getElementById("dateChance").innerText =
"Probability Meghal should go to a cute café today: " + percent + "%"

addPoints(1)

}

function chat(){

let input = document.getElementById("chatInput").value.toLowerCase()
let response = ""

if(input.includes("bored"))
response="Meghal should spin the adventure wheel 🎡"

else if(input.includes("coffee"))
response="AI recommendation: cute café + fresh juice ☕🍹"

else if(input.includes("travel"))
response="Destination suggestion: Bali 🌴"

else if(input.includes("hello"))
response="Hello Meghal 👋 Ready for an adventure?"

else
response="AI is still learning about Meghal 😄 Try asking about travel or coffee."

document.getElementById("chatBox").innerText=response

}
